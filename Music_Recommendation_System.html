def preprocess(text):
  text=re.sub(r'[^a-zA-Z]',' ',text)
  text=text.lower()
  text=word_tokenize(text)
  text=[word for word in text if word not in stop_words]
  return ' '.join(text)

#Recommendation Function
def recommend(song_name,cosine_sim=cosine_sim,new_df=new_df,top_n=5):
  index_of_song=new_df[new_df['song'].str.lower()==song_name.lower()].index
  if len(index_of_song)==0:
    return "Song not found"
  index_of_song=index_of_song[0]

  #get similarity scores
  sim_scores=list(enumerate(cosine_sim[index_of_song]))
  sim_scores=sorted(sim_scores,key=lambda x:x[1],reverse=True)
  sim_scores=sim_scores[1:top_n+1]

  song_indices=[i[0] for i in sim_scores]
  return new_df[['artist','song']].iloc[song_indices]
